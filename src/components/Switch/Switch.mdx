A switch is simply an alternative to your typical Checkbox

```typescript jsx
const Example = () => {
  const [isChecked, setChecked] = React.useState(false);
  return (
    <Box>
      <Box mb={3}>
        <Switch
          checked={isChecked}
          onChange={e => setChecked(e.target.checked)}
          aria-label="Without Label"
        />
      </Box>
      <Box mb={3}>
        <Switch
          checked={isChecked}
          onChange={e => setChecked(e.target.checked)}
          label="With Label"
        />
      </Box>
      <Box mb={3}>
        <Switch
          checked={isChecked}
          onChange={e => setChecked(e.target.checked)}
          label="With Custom names"
          checkedText="YES"
          uncheckedText="NO"
        />
      </Box>
      <Box mb={3}>
        <Switch
          invalid
          checked={isChecked}
          onChange={e => setChecked(e.target.checked)}
          label="Invalid"
        />
      </Box>
      <Box mb={3}>
        <Switch
          checked={isChecked}
          onChange={e => setChecked(e.target.checked)}
          disabled
          label="Disabled"
        />
      </Box>
    </Box>
  );
};

<Example />;
```
