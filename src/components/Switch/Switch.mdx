A switch is simply an alternative to your typical Checkbox

```typescript jsx
const Example = () => {
  const [isChecked, setChecked] = React.useState(false);
  return (
    <Box backgroundColor="navyblue-600" width="100%" p={4}>
      <Box mb={3}>
        <Switch checked={isChecked} onChange={setChecked} aria-label="Without Label" />
      </Box>
      <Box mb={3}>
        <Switch checked={isChecked} onChange={setChecked} label="With Label" />
      </Box>
      <Box mb={3}>
        <Switch
          checked={isChecked}
          onChange={setChecked}
          label="With Custom names"
          checkedText="YES"
          uncheckedText="NO"
        />
      </Box>
      <Box mb={3}>
        <Switch invalid checked={isChecked} onChange={setChecked} label="Invalid" />
      </Box>
      <Box mb={3}>
        <Switch checked={isChecked} onChange={setChecked} disabled label="Disabled" />
      </Box>
    </Box>
  );
};

<Example />;
```
