As with every other form component, the DateRangeInput is a controlled component and so it
requires for you to specify the `value` and the `onChange` handler.

Any extra prop that you specify will be given to the underlying input element of the module.

---

```typescript jsx
const Example = () => {
  const [date, setDate] = React.useState({
    start: new Date(),
    end: new Date(Date.now() + 12096e5),
  });

  const onChange = React.useCallback(changed => {
    setDate(changed);
  });
  return (
    <Box>
      <Box mb={2}>
        From: {date.start.toLocaleDateString()} to {date.end.toLocaleDateString()}
      </Box>
      <Box>
        <DateRangeInput onChange={setDate} start={date.start} end={date.end} id="hello" />
      </Box>
    </Box>
  );
};

<Example />;
```

```typescript jsx
const Example = () => {
  const format = 'MM/DD/YYYY hh:mm A';
  const [date, setDate] = React.useState({
    start: new Date(),
    end: new Date(Date.now() + 12096e5),
  });

  const onChange = React.useCallback(changed => {
    setDate(changed);
  });
  return (
    <Box>
      <Box mb={2}>
        From: {date.start.toLocaleDateString()} {date.start.toLocaleTimeString()} to{' '}
        {date.end.toLocaleDateString()} {date.end.toLocaleTimeString()}
      </Box>
      <Box>
        <DateRangeInput
          format={format}
          onChange={setDate}
          start={date.start}
          end={date.end}
          id="hello-world"
          withTime
        />
      </Box>
    </Box>
  );
};

<Example />;
```
