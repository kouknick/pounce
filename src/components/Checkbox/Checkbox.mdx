Your bread and butter Checkbox.

```typescript jsx
const Example = () => {
  const [isCheckedA, setCheckedA] = React.useState(false);
  const [isCheckedB, setCheckedB] = React.useState(false);
  const [isCheckedC, setCheckedC] = React.useState(false);
  return (
    <Box backgroundColor="navyblue-600" width="100%" height={200}>
      <Box>
        <Checkbox checked={isCheckedA} onChange={setCheckedA} />
      </Box>
      <Box>
        <Checkbox checked={isCheckedB} onChange={setCheckedB} label="With Label" />
      </Box>
      <Box>
        <Checkbox checked={isCheckedC} onChange={setCheckedC} label="Disabled" disabled />
      </Box>
    </Box>
  );
};

<Example />;
```
