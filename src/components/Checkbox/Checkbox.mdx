const Example = () => {
  const [isCheckedA, setCheckedA] = React.useState(false);
  const [isCheckedB, setCheckedB] = React.useState(false);
  const [isCheckedC, setCheckedC] = React.useState(false);
  const [isCheckedD, setCheckedD] = React.useState(false);
  return (
    <Box>
      <Box>
        <Checkbox
          checked={isCheckedA}
          onChange={e => setCheckedA(e.target.checked)}
          aria-label="Without Label"
        />
      </Box>
      <Box>
        <Checkbox
          checked={isCheckedB}
          onChange={e => setCheckedB(e.target.checked)}
          label="With Label"
        />
      </Box>
      <Box>
        <Checkbox
          checked={isCheckedC}
          onChange={e => setCheckedC(e.target.checked)}
          label="Invalid"
          invalid
        />
      </Box>
      <Box>
        <Checkbox
          checked={isCheckedD}
          onChange={e => setCheckedD(e.target.checked)}
          label="Disabled"
          disabled
        />
      </Box>
    </Box>
  );
};

<Example />;
