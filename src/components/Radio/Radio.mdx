Your bread and butter Checkbox.

```typescript jsx
const Example = () => {
  const [checked, setChecked] = React.useState('A');
  return (
    <Box backgroundColor="navyblue-600" width="100%" height={200}>
      <Box>
        <Radio checked={checked === 'A'} onChange={() => setChecked('A')} value="A" />
      </Box>
      <Box>
        <Radio
          checked={checked === 'B'}
          onChange={() => setChecked('B')}
          value="B"
          label="With Label"
        />
      </Box>
      <Box>
        <Radio
          checked={checked === 'C'}
          onChange={() => setChecked('C')}
          value="C"
          label="Disabled"
          disabled
        />
      </Box>
    </Box>
  );
};

<Example />;
```
